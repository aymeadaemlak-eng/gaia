(function() {
    const state = window.__FLOW_BUILDER_STATE__ || { customFields: [], actions: [], toggles: new Set(), savedRange: null };
    window.__FLOW_BUILDER_STATE__ = state;

    const palettePanel = document.getElementById('palettePanel');
    const paletteCustom = document.getElementById('paletteCustom');
    const paletteActions = document.getElementById('paletteActions');
    const paletteSearch = document.getElementById('paletteSearch');

    const paletteInner = palettePanel ? palettePanel.querySelector('.palette-panel') : null;

    let paletteItems = [];

    function hydrateFromBubbleConfig() {
        let cfg = {};
        const optsEl = document.getElementById('bubble-options');
        if (optsEl) {
            try {
                cfg = JSON.parse(optsEl.textContent || '{}');
            } catch (err) {
                console.warn('bubble-options parse error', err);
            }
        }

        const winCfg = window.BUBBLE_CONFIG || {};
        cfg = {
            editableHTML: winCfg.editableHTML || cfg.editableHTML || '',
            customFields: (Array.isArray(winCfg.customFields) && winCfg.customFields.length)
                ? winCfg.customFields
                : (Array.isArray(cfg.customFields) ? cfg.customFields : []),
            actions: (Array.isArray(winCfg.actions) && winCfg.actions.length)
                ? winCfg.actions
                : (Array.isArray(cfg.actions) ? cfg.actions : []),
            toggles: (winCfg.toggles && Object.keys(winCfg.toggles || {}).length)
                ? winCfg.toggles
                : (cfg.toggles || {})
        };

        const editableTemplate = document.getElementById('bubble-editable-template');
        const editableHTML = cfg.editableHTML || (editableTemplate ? editableTemplate.innerHTML.trim() : '');
        if (editableText && editableHTML) {
            editableText.innerHTML = editableHTML;
        }

        state.customFields = Array.isArray(cfg.customFields) ? cfg.customFields : [];
        state.actions = Array.isArray(cfg.actions) ? cfg.actions : [];
        const togglesCfg = cfg.toggles && typeof cfg.toggles === 'object' ? cfg.toggles : {};
        state.toggles = new Set(
            Object.entries(togglesCfg)
                .filter(([, value]) => !!value)
                .map(([key]) => key)
        );
    }

    function applyToggles() {
        state.toggles.forEach(toggle => {
            const el = document.getElementById(`toggle${toggle.charAt(0).toUpperCase() + toggle.slice(1)}`);
            if (el) el.classList.add('active');
        });
    }

            function createToken(content, className, dataId) {
                const token = document.createElement('span');
        token.classList.add('token');
        if (className) token.classList.add(className);
                token.textContent = content;
                token.dataset.id = dataId;
                token.onclick = () => token.remove();
                token.contentEditable = "false";
                token.draggable = false;
                return token;
            }

    function insertTokenFromItem(item, fallbackClass = 'action') {
        if (!item) return;
        const className = item.class || fallbackClass;
        const label = item.label || '';
        const id = item.id || '';
        if (!label) return;
        const token = createToken(label.split(' (')[0], className, id);
                const space = document.createTextNode(' ');

                const selection = window.getSelection();
        let range = selection?.rangeCount ? selection.getRangeAt(0) : null;

        if (!(range && editableText.contains(range.commonAncestorContainer))) {
            if (state.savedRange) {
                range = state.savedRange.cloneRange();
            } else {
                range = document.createRange();
                range.selectNodeContents(editableText);
                range.collapse(false);
            }
        }

        if (range && editableText.contains(range.commonAncestorContainer)) {
                    range.deleteContents();
                    range.insertNode(space.cloneNode());
                    range.insertNode(token);
                    range.setStartAfter(token);
                    selection.removeAllRanges();
                    selection.addRange(range);
                } else {
            editableText.appendChild(space);
            editableText.appendChild(token);
                }

        editableText.focus();
            }

    function currentRangeRect() {
        const sel = window.getSelection();
        let range = sel?.rangeCount ? sel.getRangeAt(0) : null;
        if (!(range && editableText.contains(range.commonAncestorContainer))) {
            range = state.savedRange;
        }
        if (!range) return null;
        const rects = range.getClientRects();
        const rect = rects.length ? rects[rects.length - 1] : range.getBoundingClientRect();
        return rect;
    }

    function positionFloatingTrigger() {
        if (!caretTrigger || !container) return;
        const rect = currentRangeRect();
        if (!rect) {
            caretTrigger.style.display = 'none';
            return;
        }
        const cRect = container.getBoundingClientRect();
        const top = rect.top - cRect.top + container.scrollTop - 6;
        const left = rect.right - cRect.left + container.scrollLeft + 8;
        caretTrigger.style.top = `${top}px`;
        caretTrigger.style.left = `${left}px`;
        caretTrigger.style.display = 'flex';
            }

            function getAllElements(className) {
        if (!editableText) {
                    console.error('Editable container not found.');
                    return [];
                }
        const elements = editableText.querySelectorAll(`.${className}`);
                return Array.from(elements).map(element => element.dataset.id);
            }

            function savePrompt() {
        const promptContentText = editableText.innerText.trim();
        const promptContentHTML = editableText.innerHTML.trim();

                const mandatoryFieldsOutput = getAllElements('mandatory');
                const optionalFieldsOutput = getAllElements('optional');
                const controlFieldsOutput = getAllElements('control');
                const actionsOutput = getAllElements('action');
        const memoryOutput = state.toggles.has('memory');
        const autoflowOutput = state.toggles.has('autoflow');

                if (typeof bubble_fn_savePrompt === 'function') {
                    bubble_fn_savePrompt({
                        output1: promptContentText,
                        output2: promptContentHTML,
                        outputlist1: mandatoryFieldsOutput,
                        outputlist2: optionalFieldsOutput,
                        outputlist3: controlFieldsOutput,
                        outputlist4: actionsOutput,
                        output3: autoflowOutput,
                        output4: memoryOutput
                    });
                }
            }

            function testPrompt() {
        const promptContentText = editableText.innerText.trim();
                const promptContentHTML = "Test: This is a special trigger text.";

                const mandatoryFieldsOutput = getAllElements('mandatory');
                const optionalFieldsOutput = getAllElements('optional');
                const controlFieldsOutput = getAllElements('control');
                const actionsOutput = getAllElements('action');
        const memoryOutput = state.toggles.has('memory');
        const autoflowOutput = state.toggles.has('autoflow');

                if (typeof bubble_fn_testPrompt === 'function') {
                    bubble_fn_testPrompt({
                        output1: promptContentText,

                        outputlist1: mandatoryFieldsOutput,
                        outputlist2: optionalFieldsOutput,
                        outputlist3: controlFieldsOutput,
                        outputlist4: actionsOutput,
                        output3: autoflowOutput,
                        output4: memoryOutput
                    });
                }
    }

    function toggleOption(option) {
        if (state.toggles.has(option)) {
            state.toggles.delete(option);
        } else {
            state.toggles.add(option);
        }
        const toggleElement = document.getElementById(`toggle${option.charAt(0).toUpperCase() + option.slice(1)}`);
        if (toggleElement) toggleElement.classList.toggle('active', state.toggles.has(option));
    }
    window.toggleOption = toggleOption;
    window.savePrompt = savePrompt;
    window.testPrompt = testPrompt;

    function renderPalette(filterText = '') {
        const q = filterText.trim().toLowerCase();
        paletteItems = [];
        selectedIndex = -1;

        if (paletteCustom) {
            paletteCustom.innerHTML = '';
            state.customFields
                .filter(item => (item.label || '').toLowerCase().includes(q))
                .forEach(item => {
                    const btn = document.createElement('button');
                    btn.type = 'button';
                    btn.className = 'palette-item';
                    const label = document.createElement('span');
                    label.textContent = item.label || 'Untitled';
                    const badge = document.createElement('span');
                    badge.className = 'palette-badge';
                    badge.textContent = item.class || 'field';
                    btn.append(label, badge);
                    btn.onclick = () => { insertTokenFromItem(item, item.class || 'optional'); hidePalette(); };
                    paletteCustom.appendChild(btn);
                    paletteItems.push({ type: 'field', data: item, btn });
                });
        }

        if (paletteActions) {
            paletteActions.innerHTML = '';
            state.actions
                .filter(item => (item.label || '').toLowerCase().includes(q))
                .forEach(item => {
                    const btn = document.createElement('button');
                    btn.type = 'button';
                    btn.className = 'palette-item';
                    const label = document.createElement('span');
                    label.textContent = item.label || 'Action';
                    const badge = document.createElement('span');
                    badge.className = 'palette-badge';
                    badge.textContent = item.class || 'action';
                    btn.append(label, badge);
                    btn.onclick = () => { insertTokenFromItem(item, item.class || 'action'); hidePalette(); };
                    paletteActions.appendChild(btn);
                    paletteItems.push({ type: 'action', data: item, btn });
                });
        }

        if (paletteItems.length) {
            selectedIndex = 0;
            updateSelection();
        }
    }

    function updateSelection() {
        paletteItems.forEach((item, idx) => {
            item.btn.classList.toggle('is-active', idx === selectedIndex);
        });
    }

    function moveSelection(delta) {
        if (!paletteItems.length) return;
        selectedIndex = (selectedIndex + delta + paletteItems.length) % paletteItems.length;
        updateSelection();
    }

    function activateSelection() {
        if (selectedIndex < 0 || !paletteItems[selectedIndex]) return;
        const sel = paletteItems[selectedIndex];
        insertTokenFromItem(sel.data, sel.type === 'field' ? (sel.data.class || 'optional') : (sel.data.class || 'action'));
        hidePalette();
    }

    function positionPalette() {
        if (!palettePanel || !paletteInner || !container) return;
        const rect = currentRangeRect();
        if (!rect) return;
        const containerRect = container.getBoundingClientRect();
        const top = Math.max(0, rect.bottom - containerRect.top + container.scrollTop + 8);
        const left = Math.max(0, rect.left - containerRect.left + container.scrollLeft);
        palettePanel.style.top = '0';
        palettePanel.style.left = '0';
        paletteInner.style.top = `${top}px`;
        paletteInner.style.left = `${left}px`;
    }

    function showPalette(focusSearch = true) {
        const sel = window.getSelection();
        if (sel?.rangeCount && editableText.contains(sel.getRangeAt(0).commonAncestorContainer)) {
            state.savedRange = sel.getRangeAt(0).cloneRange();
        } else {
            const r = document.createRange();
            r.selectNodeContents(editableText);
            r.collapse(false);
            state.savedRange = r;
        }

        renderPalette('');
        positionPalette();
        palettePanel?.classList.add('active');
        paletteInner?.classList.add('active');
        if (paletteSearch && focusSearch) {
            paletteSearch.value = '';
            paletteSearch.focus();
        }
    }

    function hidePalette() {
        palettePanel?.classList.remove('active');
        paletteInner?.classList.remove('active');
        selectedIndex = -1;
    }

    caretTrigger?.addEventListener('click', () => {
        if (palettePanel?.classList.contains('active')) hidePalette();
        else showPalette(true);
    });

    editableText?.addEventListener('keydown', (e) => {
        if (e.key === '/') {
            e.preventDefault();
            showPalette(true);
        } else if (e.key === 'Escape') {
            hidePalette();
        }
    });

    paletteSearch?.addEventListener('input', (e) => {
        renderPalette(e.target.value || '');
        positionPalette();
    });

    paletteSearch?.addEventListener('keydown', (e) => {
        if (e.key === 'ArrowDown') {
            e.preventDefault();
            moveSelection(1);
        } else if (e.key === 'ArrowUp') {
            e.preventDefault();
            moveSelection(-1);
        } else if (e.key === 'Enter') {
            e.preventDefault();
            activateSelection();
        } else if (e.key === 'Escape') {
            hidePalette();
            editableText?.focus();
        }
    });

    ['keyup','mouseup','input','focus'].forEach(evt => {
        editableText?.addEventListener(evt, () => {
            positionFloatingTrigger();
        });
    });
    editableText?.addEventListener('blur', () => {
        if (caretTrigger) caretTrigger.style.display = 'none';
    });
    document.addEventListener('selectionchange', () => {
        positionFloatingTrigger();
    });

    hydrateFromBubbleConfig();
    applyToggles();
    renderPalette();
    positionFloatingTrigger();

    document.addEventListener('DOMContentLoaded', () => {
        hydrateFromBubbleConfig();
        applyToggles();
        renderPalette();
        positionFloatingTrigger();
    });

    setTimeout(() => {
        if (!document.getElementById('editableText')?.innerHTML.trim()) {
            hydrateFromBubbleConfig();
            applyToggles();
            renderPalette();
            positionFloatingTrigger();
        }
    }, 200);
})();
